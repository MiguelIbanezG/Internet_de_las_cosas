default
{
    state_entry()
    {
        // Inicia un sensor para detectar avatares en un radio de 10 metros
        llSensorRepeat("", "", AGENT, 10.0, PI, 2.0); 
    }

    sensor(integer total_number)
    {
        if (total_number > 0)
        {
            string name = llDetectedName(0); // Obtiene el nombre del primer avatar detectado
            llSetText("Hola " + name, <1,1,1>, 1.0); // Muestra el mensaje con color blanco y opacidad total
        }
    }
    
    no_sensor()
    {
        llSetText("", <1,1,1>, 1.0); // Borra el texto cuando no hay avatares cerca
    }
}
